from pathlib import Path

script = r'''#!/usr/bin/env bash

set -euo pipefail

osascript <<'APPLESCRIPT'

tell application "System Events"

  tell process "ChatGPT"

    set allItems to entire contents of window 1

    repeat with itemRef in allItems

      set roleText to ""

      set descText to ""

      try

        set roleText to role of itemRef as text

      end try

      try

        set descText to description of itemRef as text

      end try

      if roleText is "AXStaticText" and descText is not "" and descText is not "missing value" then

        if length of descText > 1600 then set descText to text 1 thru 1600 of descText

        log descText

      end if

    end repeat

  end tell

end tell

APPLESCRIPT

'''

path = Path("bin/pinky-read-visible")

path.write_text(script, encoding="utf-8")

path.chmod(0o755)

print("created bin/pinky-read-visible")

from pathlib import Path

script = r'''#!/usr/bin/env bash

set -euo pipefail

TMP_FILE="$(mktemp)"

trap 'rm -f "$TMP_FILE"' EXIT

"$PWD/bin/pinky-read-visible" 2>&1 > "$TMP_FILE" || {

  echo "pinky-detect-signal error: failed to read visible ChatGPT text" >&2

  exit 1

}

if grep -q "🛑" "$TMP_FILE"; then

  echo "signal=stop"

  exit 0

fi

if grep -q "🧠" "$TMP_FILE"; then

  echo "signal=brain"

  exit 0

fi

if grep -q "🐭" "$TMP_FILE"; then

  echo "signal=pinky"

  exit 0

fi

echo "signal=none"

'''

path = Path("bin/pinky-detect-signal")

path.write_text(script, encoding="utf-8")

path.chmod(0o755)

print("created bin/pinky-detect-signal")
