We found the bug: pinky-detect-signal was not actually capturing osascript log output because the redirection order was wrong.

Patch it so it captures both stdout/stderr, then chooses the latest visible signal by line order:
cd ~/Desktop/pinky-brain && python3 - <<'PY'

from pathlib import Path

script = r'''#!/usr/bin/env bash

set -euo pipefail

ROOT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"

TMP_FILE="$(mktemp)"

trap 'rm -f "$TMP_FILE"' EXIT

"$ROOT_DIR/bin/pinky-read-visible" > "$TMP_FILE" 2>&1 || {

  echo "pinky-detect-signal error: failed to read visible ChatGPT text" >&2

  exit 1

}

latest="none"

while IFS= read -r line; do

  case "$line" in

    *"🐭"*) latest="pinky" ;;

  esac

  case "$line" in

    *"🧠"*) latest="brain" ;;

  esac

  case "$line" in

    *"🛑"*) latest="stop" ;;

  esac

done < "$TMP_FILE"

echo "signal=$latest"

'''

path = Path("bin/pinky-detect-signal")

path.write_text(script, encoding="utf-8")

path.chmod(0o755)

print("patched bin/pinky-detect-signal capture/order logic")
